## 系统概述

### 设计理念
针对 2D ARPG 游戏特点，设计了一套完整的物品管理系统：
- **背包系统**: 基于格子的物品存储，支持自动堆叠、重量限制、智能整理
- **装备管理器**: 11个装备槽位，套装系统，耐久度管理，无缝集成属性系统

### 核心特性
✅ **格子式背包** - 直观的网格布局，支持拖拽操作  
✅ **智能堆叠** - 相同物品自动合并，优化背包空间  
✅ **重量系统** - 可选的负重限制，增加策略性  
✅ **装备槽位** - 11个独立槽位（头、胸、腿、靴、手套、武器、副手、戒指×2、项链、腰带）  
✅ **套装系统** - 自动检测套装件数，触发套装效果  
✅ **耐久度** - 可选的装备耐久度消耗与修理  
✅ **序列化** - 完整的存档/读档支持

## 装备管理器

### 类: EquipmentManager

#### 核心功能

##### 1. 装备操作
```gdscript
# 装备物品
func equip(item: ItemInstance, auto_swap: bool = true) -> bool

# 卸下装备
func unequip(slot: EquipmentData.EquipSlot, to_inventory: bool = true) -> bool

# 从背包直接装备
func equip_from_inventory(slot_index: int) -> bool
```

**装备流程**:
1. 检查物品类型和等级需求
2. 处理当前槽位装备（自动卸下或返回失败）
3. 应用属性修正器到 StatsComponent
4. 更新套装计数
5. 处理绑定逻辑

##### 2. 套装系统
```gdscript
# 获取套装件数
func get_set_piece_count(set_id: String) -> int

# 检查套装效果
func has_set_bonus(set_id: String, required_pieces: int) -> bool

# 获取所有激活套装
func get_active_sets() -> Array
```

**套装检测**:
- 自动追踪每个套装ID的装备数量
- 达到阈值（2/4/6件）时发出信号
- 支持多套装同时激活

##### 3. 耐久度管理
```gdscript
# 受击消耗耐久度（随机选择一件装备）
func consume_durability_on_hit() -> void

# 消耗指定槽位耐久度
func consume_durability(slot: EquipmentData.EquipSlot, amount: int = 1) -> void

# 修理装备
func repair_equipment(slot: EquipmentData.EquipSlot, full_repair: bool = true) -> int

# 修理所有装备
func repair_all() -> int
```

**耐久度特性**:
- 可配置的消耗概率和数值
- 耐久度为0时装备失效（可选移除属性加成）
- 计算修理费用

#### 关键信号

### 2. UI集成建议

#### 背包UI
```gdscript
class_name InventoryUI extends Control

@onready var inventory: InventoryComponent = get_node("../../InventoryComponent")
var slot_buttons: Array[Button] = []

func _ready():
    # 连接信号
    inventory.slot_changed.connect(_update_slot_ui)
    inventory.gold_changed.connect(_update_gold_display)
    
    # 创建格子按钮
    _create_slot_buttons()

func _create_slot_buttons():
    for i in range(inventory.slot_count):
        var button = Button.new()
        button.pressed.connect(_on_slot_clicked.bind(i))
        slot_buttons.append(button)

func _update_slot_ui(slot_index: int):
    var item = inventory.get_item(slot_index)
    if item:
        slot_buttons[slot_index].icon = item.item_data.icon
        slot_buttons[slot_index].text = str(item.stack_count)
    else:
        slot_buttons[slot_index].icon = null
        slot_buttons[slot_index].text = ""

func _on_slot_clicked(slot_index: int):
    var item = inventory.get_item(slot_index)
    if item:
        # 显示物品信息或使用/装备
        if Input.is_action_pressed("shift"):
            # Shift+点击 = 快速使用/装备
            _quick_use(slot_index)
        else:
            # 普通点击 = 显示详情
            _show_item_details(item)
```

#### 装备UI
```gdscript
class_name EquipmentUI extends Control

@onready var equipment: EquipmentManager = get_node("../../EquipmentManager")
var slot_displays: Dictionary = {}

func _ready():
    equipment.equipment_changed.connect(_update_slot_display)
    equipment.set_bonus_activated.connect(_show_set_bonus)
    
    _create_equipment_slots()

func _create_equipment_slots():
    for slot in EquipmentData.EquipSlot.values():
        var panel = Panel.new()
        var texture_rect = TextureRect.new()
        panel.add_child(texture_rect)
        slot_displays[slot] = texture_rect

func _update_slot_display(slot, item):
    var texture_rect = slot_displays[slot]
    if item:
        texture_rect.texture = item.item_data.icon
    else:
        texture_rect.texture = null
```

### 3. 拖拽系统实现

```gdscript
# 物品拖拽
class ItemDragPreview extends Control:
    var item: ItemInstance
    var source_slot: int
    
    func _ready():
        mouse_filter = MOUSE_FILTER_IGNORE
    
    func _process(_delta):
        global_position = get_global_mouse_position()

# 在 InventoryUI 中
var dragging_item: ItemDragPreview = null

func _on_slot_drag_start(slot_index: int):
    var item = inventory.get_item(slot_index)
    if item:
        dragging_item = ItemDragPreview.new()
        dragging_item.item = item
        dragging_item.source_slot = slot_index
        add_child(dragging_item)

func _on_slot_drag_end(target_slot: int):
    if dragging_item:
        inventory.move_item(dragging_item.source_slot, target_slot)
        dragging_item.queue_free()
        dragging_item = null
```

### 4. 性能优化

#### 批量操作
```gdscript
# ❌ 不好的做法 - 多次单独添加
for item in loot_items:
    inventory.add_item(item)  # 每次都触发信号和UI更新

# ✅ 好的做法 - 批量添加
var results = inventory.add_items(loot_items)  # 一次性处理
```

#### 信号优化
```gdscript
# 延迟UI更新，避免频繁刷新
var _pending_updates: Array[int] = []
var _update_timer: Timer

func _on_slot_changed(slot_index: int):
    if slot_index not in _pending_updates:
        _pending_updates.append(slot_index)
    
    if not _update_timer.is_stopped():
        return
    
    _update_timer.start(0.1)  # 100ms后批量更新

func _on_update_timer_timeout():
    for slot in _pending_updates:
        _update_slot_ui(slot)
    _pending_updates.clear()
```

## 使用示例

### 基础流程示例

#### 玩家拾取物品
```gdscript
func _on_loot_pickup(dropped_item: DroppedItem):
    var item_instance = dropped_item.item_instance
    
    if inventory.add_item(item_instance):
        print("拾取: %s" % item_instance.item_data.item_name)
        dropped_item.queue_free()
    else:
        print("背包已满!")
```

#### 装备武器
```gdscript
func equip_weapon_from_inventory(slot_index: int):
    var item = inventory.get_item(slot_index)
    if not item or not item.item_data is WeaponData:
        return
    
    # 移除背包中的物品
    var removed = inventory.remove_item(slot_index)
    
    # 尝试装备
    if equipment_manager.equip(removed):
        print("装备成功!")
    else:
        # 装备失败，放回背包
        inventory.add_item(removed)
```

#### 使用消耗品
```gdscript
func use_consumable(slot_index: int):
    var item = inventory.get_item(slot_index)
    if not item or not item.item_data is ConsumableData:
        return
    
    var consumable = item.item_data as ConsumableData
    
    # 执行效果
    match consumable.effect_type:
        ConsumableData.EffectType.INSTANT_HEAL:
            stats_component.heal(consumable.effect_value)
        ConsumableData.EffectType.BUFF:
            for mod in consumable.temp_modifiers:
                mod.set_duration(consumable.effect_duration)
                stats_component.add_modifier(mod)
    
    # 消耗物品
    inventory.remove_item(slot_index, 1)
```

## 性能优化

### 1. 缓存优化
```gdscript
# 缓存物品查找结果
var _item_cache: Dictionary = {}  # {item_id: Array[slot_index]}

func _on_item_added(item, slot):
    if item.item_data.id not in _item_cache:
        _item_cache[item.item_data.id] = []
    _item_cache[item.item_data.id].append(slot)

# 快速查找物品位置
func find_item_fast(item_id: String) -> Array:
    return _item_cache.get(item_id, [])
```

### 2. 避免不必要的计算
```gdscript
# ❌ 每帧计算重量
func _process(_delta):
    var weight = _calculate_total_weight()  # 性能浪费

# ✅ 只在物品变化时更新
func _on_slot_changed(slot):
    _update_weight()  # 只在必要时计算
```

### 3. 延迟加载
```gdscript
# 大型物品数据库延迟加载
var _item_database: Dictionary = {}
var _loaded_categories: Array[String] = []

func get_item_data(item_id: String) -> ItemData:
    if item_id not in _item_database:
        _load_item_category(item_id.get_slice("_", 0))
    return _item_database.get(item_id)
```

## 总结

这套背包与装备系统设计遵循以下原则：

✅ **模块化** - 各系统职责清晰，易于维护  
✅ **可扩展** - 支持自定义物品类型和效果  
✅ **性能优先** - 缓存、批量操作、信号优化  
✅ **用户体验** - 自动堆叠、智能整理、拖拽支持  
✅ **数据安全** - 完整的序列化和验证机制  

适用于各种 2D ARPG 项目，从简单的 RPG 到复杂的 Roguelike 游戏。
