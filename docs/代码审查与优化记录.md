### 4. 【中等】信号连接可能重复
**位置**: `scripts/quest/task_manager.gd:105-108`

**问题**:
```gdscript
# 连接信号
if not instance.state_changed.is_connected(_on_task_state_changed):
	instance.state_changed.connect(_on_task_state_changed.bind(task_id))
if not instance.progress_updated.is_connected(_on_task_progress_updated):
	instance.progress_updated.connect(_on_task_progress_updated.bind(task_id))
```

**分析**: 
- 使用 `bind()` 后，`is_connected()` 检查可能失效
- 每次接取任务都会创建新的绑定连接

**建议**: 改用 Callable 或在连接前先断开
```gdscript
# 方案1: 先断开再连接
if instance.state_changed.is_connected(_on_task_state_changed):
	instance.state_changed.disconnect(_on_task_state_changed)
instance.state_changed.connect(_on_task_state_changed.bind(task_id))

# 方案2: 使用字典存储连接状态
var _task_connections: Dictionary = {}

func _connect_task_signals(task_id: String, instance: TaskInstance) -> void:
	if _task_connections.has(task_id):
		return
	
	instance.state_changed.connect(_on_task_state_changed.bind(task_id))
	instance.progress_updated.connect(_on_task_progress_updated.bind(task_id))
	_task_connections[task_id] = true
```

### 5. 【中等】目标初始化时信号断开不完整
**位置**: `scripts/quest/task_instance.gd:44-58`

**问题**:
```gdscript
func _initialize_objectives() -> void:
	# 先断开旧的信号连接
	for obj in objectives:
		if obj.objective_completed.is_connected(_on_objective_completed):
			obj.objective_completed.disconnect(_on_objective_completed)
		if obj.objective_progress_changed.is_connected(_on_objective_progress_changed):
			obj.objective_progress_changed.disconnect(_on_objective_progress_changed)
	
	objectives.clear()
	for obj_data in task_data.objectives:
		var obj = obj_data.duplicate(true)  # 这里可能有问题
```

**问题分析**:
1. `duplicate(true)` 对 Resource 的深拷贝可能不完整
2. 信号连接状态可能无法正确复制

**建议**: 使用工厂方法创建新实例
```gdscript
func _initialize_objectives() -> void:
	# 断开旧信号
	for obj in objectives:
		if obj.objective_completed.is_connected(_on_objective_completed):
			obj.objective_completed.disconnect(_on_objective_completed)
		if obj.objective_progress_changed.is_connected(_on_objective_progress_changed):
			obj.objective_progress_changed.disconnect(_on_objective_progress_changed)
	
	objectives.clear()
	
	# 创建新实例而不是复制
	for obj_data in task_data.objectives:
		var obj: TaskObjective
		
		# 根据类型创建新实例
		if obj_data is CountObjective:
			obj = CountObjective.new()
			obj.target_type = obj_data.target_type
			obj.target_id = obj_data.target_id
			obj.required_count = obj_data.required_count
		elif obj_data is StateObjective:
			obj = StateObjective.new()
			obj.state_type = obj_data.state_type
			obj.target_state = obj_data.target_state
		else:
			obj = obj_data.duplicate(true)
		
		# 复制基类属性
		obj.objective_id = obj_data.objective_id
		obj.description = obj_data.description
		obj.optional = obj_data.optional
		obj.weight = obj_data.weight
		
		obj.initialize()
		obj.objective_completed.connect(_on_objective_completed)
		obj.objective_progress_changed.connect(_on_objective_progress_changed)
		objectives.append(obj)
```

**建议**: 考虑暂停/恢复场景

var paused_duration: float = 0.0
var pause_start_time: float = 0.0

func get_remaining_time() -> float:
	if task_data.time_limit <= 0.0:
		return -1.0
	
	if state != TaskState.State.ACTIVE:
		return task_data.time_limit
	
	var current_time = Time.get_unix_time_from_system()

#### 1.2 优化事件分发
```gdscript
# TaskManager 中添加事件过滤
var _event_subscriptions: Dictionary = {}  # {event_type: [task_ids]}

func accept_task(task_id: String) -> bool:
	# ... 原有代码 ...
	
	# 注册任务关注的事件类型
	for obj in instance.objectives:
		var event_type = ""
		if obj is CountObjective:
			event_type = obj.target_type
		elif obj is StateObjective:
			event_type = obj.state_type
		
		if event_type != "":
			if not _event_subscriptions.has(event_type):
				_event_subscriptions[event_type] = []
			_event_subscriptions[event_type].append(task_id)
	
	return true

func update_task_progress(event_data: Dictionary) -> void:
	var event_type = event_data.get("type", "")
	
	# 只更新订阅了此事件的任务
	var subscribed_tasks = _event_subscriptions.get(event_type, [])
	for task_id in subscribed_tasks:
		if player_tasks.has(task_id):
			var instance: TaskInstance = player_tasks[task_id]
			if instance.state == TaskState.State.ACTIVE:
				instance.update_objective_progress(event_data)
				
				if instance.is_expired():
					fail_task(task_id, "任务超时")
```
